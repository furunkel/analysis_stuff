bitcount

def bitcount(n):
    count = 0
    while n:
        # Here is the bug
        n ^= n - 1
        count += 1
    return count

levenshtein

 levenshtein(source, target):
    if source == '' or target == '':
        return len(source) or len(target)

    elif source[0] == target[0]:
        # Here is the bug
        return 1 + levenshtein(source[1:], target[1:])

    else:
        return 1 + min(
            levenshtein(source,     target[1:]),
            levenshtein(source[1:], target[1:]),
            levenshtein(source[1:], target)
        )
